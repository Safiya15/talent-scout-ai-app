Supabase queries needed (all filtered by user_id = auth.user.id):
javascript// 1. All candidates
const { data: candidates } = await supabase
  .from('candidates')
  .select('*')
  .eq('user_id', user.id)

// 2. Email logs for activity feed  
const { data: logs } = await supabase
  .from('email_logs')
  .select('*, candidates(name)')
  .eq('user_id', user.id)
  .order('sent_at', { ascending: false })
  .limit(8)
